255  void insert_cell(int cell_id, vector < int> cluster_id_vec, Module *m, int level) {
256  if(num_levels <= level) {
257  return;

259  if(cluster_id_vec.size() < 1) {

265  tmp->macroModule = true;

269  levels[l].modules.push_back(tmp);
270  macroModulePath.push_back(tmp);

272  for(int l=1; l<num_levels; ++l) {
273  Module *tmp = macroModulePath[l-1];
274  if (l == 1) {
275  root->children.push_back(tmp);
276  tmp->parent = root;
277  } else {
278  tmp->parent = macroModulePath[l-2];
279  }
280  tmp->children.push_back(macroModulePath[l]);
281  }
282  macroModulePath[num_levels-1]->parent = macroModulePath[num_levels-2];
283  macroModulePath[num_levels-1]->children.push_back(macroModulePath[num_levels]);
284  macroModulePath[num_levels]->parent = macroModulePath[num_levels-1];
