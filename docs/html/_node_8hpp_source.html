184  void setRotation(int r) {

188 
189  setPos(-width/2,-height/2);
190  model::polygon<model::d2::point_xy<double> > tmp;
191  trans::rotate_transformer<boost::geometry::degree, double, 2, 2> rotate(rot_deg);
192  boost::geometry::transform(poly, tmp, rotate);
193  poly = tmp;
194  double otmp = wrap_orientation(orientation + r);
195  orientation = otmp;
196  setPos(tmpx, tmpy);

199  /*
200  upateCoordinates
201  Updates parameters of Node class from a geometry object
202  */
203  void updateCoordinates() {
204  if(terminal) {
205  xBy2 = xCoordinate;
206  yBy2 = yCoordinate;
207  } else {
208  boost::geometry::model::d2::point_xy<double> centroid;
209  boost::geometry::centroid(poly, centroid);
210  xBy2 = centroid.get<0>();
211  yBy2 = centroid.get<1>();
212 
213  boost::geometry::model::box< model::d2::point_xy<double> > envtmp;
214  boost::geometry::envelope(poly, envtmp);
215  envelope = envtmp;
216  Point minCorner = envelope.min_corner();
217  Point maxCorner = envelope.max_corner();
218  width = maxCorner.get<0>() - minCorner.get<0>();
219  height = maxCorner.get<1>() - minCorner.get<1>();
220  xCoordinate = bg::get<bg::min_corner, 0>(envelope);
221  yCoordinate = bg::get<bg::min_corner, 1>(envelope);
222  }
223  }
