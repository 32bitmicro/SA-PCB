140  void showUsage(const string comment, TimeType type)
141  {
142  TimeState curSt;
143  checkUsage(curSt);
144  TimeState dur = (type == FULL) ? diff(tStart_, curSt) : diff(pStart_, curSt);
145  if (type == FULL)
146  {
147  cout << "---------- " << comment << " total time usage -----------" << endl;
148  cout << " Final Real:" << dur.rTime << "s;";
149  }
150  else
151  {
152  cout << "---------- " << comment << " period time usage -----------" << endl;
153  cout << " Real:" << dur.rTime << "s;";
154  }
155  cout << " User:" << dur.uTime << "s;";
156  cout << " System:" << dur.sTime << "s." << endl
157  << endl;
158  }
159 
160 private:
161  TimeState diff(TimeState &start, TimeState &end)
162  {
163  return TimeState(end.rTime - start.rTime, end.uTime - start.uTime, end.sTime - start.sTime);
164  }
165 
166  void checkUsage(TimeState &st) const
167  {
168  rusage tUsg;
169  getrusage(RUSAGE_SELF, &tUsg);
170  timeval tReal;
171  gettimeofday(&tReal, NULL);
172  st.uTime = tUsg.ru_utime.tv_sec + tUsg.ru_utime.tv_usec / TIME_SCALE;
173  st.sTime = tUsg.ru_stime.tv_sec + tUsg.ru_stime.tv_usec / TIME_SCALE;
174  st.rTime = tReal.tv_sec + tReal.tv_usec / TIME_SCALE;
175  }
176  TimeState tStart_, pStart_; //total, period
177 };
